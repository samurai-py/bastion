```rust
use toml_edit::{DocumentMut, table, value};

/// Validate age_pubkey format before persisting. Must match ^age1[0-9a-z]+$
fn validate_age_pubkey(key: &str) -> anyhow::Result<()> {
    let re = regex::Regex::new(r"^age1[0-9a-z]+$")
        .expect("static regex must compile");
    if !re.is_match(key) {
        anyhow::bail!("invalid age_pubkey format — must match ^age1[0-9a-z]+$ (SEC-01)");
    }
    Ok(())
}

/// Append a new [[mesh.peer]] entry to bastion.toml using toml_edit.
///
/// SEC-01: uses toml_edit (programmatic table construction, no string interpolation).
/// WR-02: preserves existing entries including allowed_tags; bails on read error
///        instead of overwriting the file with empty content.
pub async fn append_mesh_peer(
    owner_id: &str,
    peer_url: &str,
    age_pubkey: &str,
    allowed_tags: &[String],
) -> anyhow::Result<()> {
    // Validate age_pubkey before touching the file
    validate_age_pubkey(age_pubkey)?;

    let path = std::env::var("BASTION_CONFIG").unwrap_or_else(|_| "bastion.toml".to_string());

    // WR-02: bail on read error — do NOT fall back to empty string (would overwrite entire config)
    let current = tokio::fs::read_to_string(&path).await
        .map_err(|e| anyhow::anyhow!("failed to read {} before appending peer: {}", path, e))?;

    // Parse as mutable TOML document (toml_edit preserves comments/formatting)
    let mut doc: DocumentMut = current.parse()
        .map_err(|e| anyhow::anyhow!("failed to parse {} as TOML: {}", path, e))?;

    // Build the new [[mesh.peer]] entry as a toml_edit Table
    let mut peer_entry = toml_edit::Table::new();
    peer_entry["owner_id"]   = value(owner_id);
    peer_entry["peer_url"]   = value(peer_url);
    peer_entry["age_pubkey"] = value(age_pubkey);
    if !allowed_tags.is_empty() {
        let tags_array: toml_edit::Array = allowed_tags.iter()
            .map(|t| toml_edit::Value::from(t.as_str()))
            .collect();
        peer_entry["allowed_tags"] = toml_edit::Item::Value(toml_edit::Value::Array(tags_array));
    }

    // Append to [[mesh.peer]] array-of-tables
    doc["mesh"]["peer"]
        .as_array_of_tables_mut()
        .map(|arr| arr.push(peer_entry.clone()));
    // If mesh.peer key doesn't exist yet, create it
    if doc["mesh"]["peer"].as_array_of_tables().is_none() {
        let mut aot = toml_edit::ArrayOfTables::new();
        aot.push(peer_entry);
        doc["mesh"]["peer"] = toml_edit::Item::ArrayOfTables(aot);
    }

    // Write to a temp file first, then atomically rename (prevents partial write corruption)
    let tmp_path = format!("{}.tmp", path);
    tokio::fs::write(&tmp_path, doc.to_string()).await?;
    tokio::fs::rename(&tmp_path, &path).await?;

    Ok(())
}
```

```rust
// SEC-02: validate peer_url before registering — prevent SSRF to loopback/RFC1918/link-local.
{
    use url::Url;

    let parsed = Url::parse(&body.peer_url)
        .map_err(|_| ())
        .and_then(|u| if u.scheme() == "https" { Ok(u) } else { Err(()) });

    let parsed = match parsed {
        Ok(u) => u,
        Err(_) => {
            tracing::warn!(event = "mesh_pair_invalid_url", url = %body.peer_url);
            return (StatusCode::BAD_REQUEST, Json(serde_json::json!({ "error": "peer_url must be a valid https:// URL" }))).into_response();
        }
    };

    // DNS-resolve and reject private/loopback/link-local addresses
    let host = parsed.host_str().unwrap_or("");
    if let Ok(addrs) = tokio::net::lookup_host(format!("{}:443", host)).await {
        for addr in addrs {
            let ip = addr.ip();
            if ip.is_loopback() || ip.is_unspecified() || is_private_ip(ip) {
                tracing::warn!(event = "mesh_pair_ssrf_blocked", url = %body.peer_url, ip = %ip);
                return (StatusCode::BAD_REQUEST, Json(serde_json::json!({ "error": "peer_url resolves to a private/loopback address" }))).into_response();
            }
        }
    }
}
```

```rust
fn is_private_ip(ip: std::net::IpAddr) -> bool {
    match ip {
        std::net::IpAddr::V4(v4) => {
            v4.is_private()       // 10.x, 172.16-31.x, 192.168.x
            || v4.is_link_local() // 169.254.x
        }
        std::net::IpAddr::V6(v6) => {
            v6.is_loopback()
            || (v6.segments()[0] & 0xfe00) == 0xfc00 // fc00::/7 ULA
        }
    }
}
```
