Change the function signature to also accept `jwt_secret`:

```rust
fn resolve_owner_or_401(
    headers: &HeaderMap,
    owner_map: &OwnerMap,
    jwt_secret: &str,
    event_name: &'static str,
) -> Result<String, axum::response::Response> {
    use jsonwebtoken::{decode, DecodingKey, Validation, Algorithm};

    let token = headers
        .get("x-bastion-token")
        .and_then(|v| v.to_str().ok())
        .unwrap_or("");

    // First try JWT decode (mobile app tokens)
    let validation = Validation::new(Algorithm::HS256);
    match decode::<Claims>(
        token,
        &DecodingKey::from_secret(jwt_secret.as_bytes()),
        &validation,
    ) {
        Ok(data) => return Ok(data.claims.sub),
        Err(_) => {
            // Fall through to static owner_map lookup (existing CLI/API tokens)
        }
    }

    // Fallback: static owner_map (pre-existing non-JWT tokens)
    match owner_map.resolve(token) {
        Some(o) => Ok(o.to_owned()),
        None => {
            tracing::warn!(event = event_name, "unknown or missing x-bastion-token");
            Err((StatusCode::UNAUTHORIZED, Json(serde_json::json!({}))).into_response())
        }
    }
}
```
