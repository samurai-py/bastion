```rust
//! P2P MeshTransport implementation (OSS).
//! Sends encrypted MeshEnvelopes to peer daemons via HTTP POST to /mesh/ingest.
//! Uses `age` (X25519 + ChaCha20-Poly1305) for E2E encryption.
//! Relay impl (Bastion Cloud, closed) is a separate repo — same MeshTransport trait.

use std::sync::Arc;
use tokio::sync::broadcast;
use opentelemetry::{global as otel_global, trace::{Tracer, SpanKind}, KeyValue};
use anyhow::Context;
use serde_json;

use crate::mesh::{MeshTransport, MeshEnvelope, MeshEnvelope, SelectiveSlice, MeshPeerMap};

pub struct P2PTransport {
    /// Local age identity private key (bech32 string). Loaded from env MESH_IDENTITY_KEY.
    local_identity_key: String,
    /// Registered peer map: owner_id → (peer_url, age_pubkey).
    peers: Arc<MeshPeerMap>,
    /// SSE broadcast channel — receives mesh_sync events (Pitfall 3: capacity=128, ignore SendError).
    events_tx: broadcast::Sender<String>,
    /// local owner_id (for from_owner in outbound envelopes).
    local_owner: String,
    /// HTTP client (reqwest, already in Cargo.toml).
    http: reqwest::Client,
}

impl P2PTransport {
    pub fn new(
        local_owner: String,
        local_identity_key: String,
        peers: Arc<MeshPeerMap>,
        events_tx: broadcast::Sender<String>,
    ) -> Self {
        Self {
            local_owner,
            local_identity_key,
            peers,
            events_tx,
            http: reqwest::Client::new(),
        }
    }
}

#[async_trait::async_trait]
impl MeshTransport for P2PTransport {
    /// Send a selective slice to a remote owner.
    ///
    /// CALLER RESPONSIBILITY: call filter_for_mesh BEFORE calling this method.
    /// This impl does NOT re-filter — it trusts the caller to have done so.
    async fn send(&self, slice: SelectiveSlice, to_owner: &str) -> anyhow::Result<()> {
        let peer = self.peers.resolve(to_owner)
            .ok_or_else(|| anyhow::anyhow!("mesh peer '{}' not registered in MeshPeerMap", to_owner))?;

        // Collect tags for OTel event BEFORE moving slice
        let tags: Vec<String> = slice.beliefs.iter()
            .filter_map(|b| b.persona_tag.clone())
            .collect::<std::collections::HashSet<_>>()
            .into_iter()
            .collect();

        // Serialize SelectiveSlice to bytes
        let plaintext = serde_json::to_vec(&slice)
            .context("failed to serialize SelectiveSlice")?;

        // age encrypt with peer's public key (X25519 + ChaCha20-Poly1305)
        let recipient: age::x25519::Recipient = peer.age_pubkey.parse()
            .map_err(|_| anyhow::anyhow!("invalid age public key for peer '{}'", to_owner))?;
        let mut ciphertext = Vec::new();
        {
            let encryptor = age::Encryptor::with_recipients(vec![Box::new(recipient)])
                .map_err(|e| anyhow::anyhow!("age encryptor error: {}", e))?;
            let mut writer = encryptor.wrap_output(
                age::armor::ArmoredWriter::wrap_output(&mut ciphertext, age::armor::Format::Binary)?
            )?;
            use std::io::Write;
            writer.write_all(&plaintext)?;
            writer.finish()?;
        }

        let envelope = MeshEnvelope {
            from_owner: self.local_owner.clone(),
            to_owner: to_owner.to_string(),
            ciphertext,
            recipient_hint: peer.age_pubkey.clone(),
        };

        // OTel mesh_sync span (SEAM #4)
        let tracer = otel_global::tracer("bastion");
        let mut sync_span = tracer
            .span_builder("mesh_sync")
            .with_kind(SpanKind::Internal)
            .with_attributes(vec![
                KeyValue::new("gen_ai.operation.name", "mesh_sync"),
                KeyValue::new("mesh.from_owner", self.local_owner.clone()),
                KeyValue::new("mesh.to_owner", to_owner.to_string()),
                KeyValue::new("mesh.tags", tags.join(",")),
                KeyValue::new("mesh.beliefs_count", slice.beliefs.len() as i64),
            ])
            .start(&tracer);

        // POST to peer /mesh/ingest
        let url = format!("{}/mesh/ingest", peer.peer_url.trim_end_matches('/'));
        let resp = self.http
            .post(&url)
            .json(&envelope)
            .send()
            .await
            .context("failed to POST to peer /mesh/ingest")?;

        let status = resp.status();
        sync_span.set_attribute(KeyValue::new("mesh.http_status", status.as_u16() as i64));
        sync_span.end();

        if !status.is_success() {
            anyhow::bail!("peer /mesh/ingest returned {}", status);
        }

        // Broadcast to SSE (Pitfall 3: ignore SendError — slow/disconnected clients are acceptable)
        let event_json = serde_json::json!({
            "type": "mesh_sync",
            "from_owner": &self.local_owner,
            "to_owner": to_owner,
            "tags": tags,
        }).to_string();
        let _ = self.events_tx.send(event_json);

        Ok(())
    }

    async fn receive(&self, envelope: MeshEnvelope) -> anyhow::Result<SelectiveSlice> {
        // age decrypt with local identity
        let identity: age::x25519::Identity = self.local_identity_key.parse()
            .map_err(|_| anyhow::anyhow!("invalid local age identity key"))?;

        let decryptor = match age::Decryptor::new(&envelope.ciphertext[..])? {
            age::Decryptor::Recipients(d) => d,
            _ => anyhow::bail!("unexpected age format in mesh envelope"),
        };
        let mut reader = decryptor
            .decrypt(std::iter::once(&identity as &dyn age::Identity))
            .map_err(|e| anyhow::anyhow!("age decrypt failed: {:?}", e))?;

        use std::io::Read;
        let mut plaintext = Vec::new();
        reader.read_to_end(&mut plaintext)?;

        let slice: SelectiveSlice = serde_json::from_slice(&plaintext)
            .context("failed to deserialize SelectiveSlice after decrypt")?;

        // Pitfall 2: verify from_owner in envelope matches decrypted slice's from_owner.
        // If a malicious sender injected a different from_owner, the slice mismatch catches it.
        if envelope.from_owner != slice.from_owner {
            anyhow::bail!(
                "from_owner mismatch: envelope claims '{}', payload contains '{}' — rejecting",
                envelope.from_owner, slice.from_owner
            );
        }

        // Verify sender is a registered peer
        if self.peers.resolve(&slice.from_owner).is_none() {
            anyhow::bail!("received mesh envelope from unregistered peer '{}'", slice.from_owner);
        }

        tracing::info!(
            event = "mesh_slice_received",
            from_owner = %slice.from_owner,
            beliefs_count = slice.beliefs.len(),
        );

        Ok(slice)
    }
}
```

**src/mesh/context_provider.rs** — replace stub with full implementation:

```rust
//! MeshSliceProvider: TurnContextProvider impl that injects a remote owner's selective
//! memory slice into the system prompt via SEAM #2 (opaque ContextBlock).
//!
//! SEAM #2 rule: content is OPAQUE — AgentLoop includes it verbatim, never parses.
//! ContextBlock.max_tier = CloudOk (mesh only carries CloudOk beliefs per filter_for_mesh).
//!
//! MESH-03 (inter-owner Cabinet) pattern:
//! Each instance runs its own Cabinet locally. The synthesis result is stored as a belief
//! tagged "mesh_cabinet_synthesis" and shared to the other owner as a SelectiveSlice.
//! This is an ASYNC exchange — Mario's turn completes with local synthesis; Ana's synthesis
//! arrives as a future mesh_sync and appears in her next context_for_turn call.
//! No blocking across instance boundaries (Pitfall 6 mitigation).

use std::sync::Arc;
use tokio::sync::{broadcast, RwLock};
use std::collections::HashMap;

use crate::agent::context::{ContextBlock, TurnContextProvider};
use crate::memory::{SharedMemory, PrivacyTier};

/// In-memory store of received slices from mesh peers.
/// Keyed by from_owner. Updated by ingest_handler when a new slice arrives.
pub type MeshSliceStore = Arc<RwLock<HashMap<String, Vec<crate::memory::Belief>>>>;

pub struct MeshSliceProvider {
    /// Receives slices from ingest_handler via this store.
    slice_store: MeshSliceStore,
    /// The local owner_id (to scope context correctly).
    local_owner: String,
}

impl MeshSliceProvider {
    pub fn new(local_owner: String) -> (Self, MeshSliceStore) {
        let store = Arc::new(RwLock::new(HashMap::new()));
        let provider = Self {
            slice_store: store.clone(),
            local_owner,
        };
        (provider, store)
    }
}

#[async_trait::async_trait]
impl TurnContextProvider for MeshSliceProvider {
    /// Inject all received mesh slices as opaque ContextBlocks.
    /// Non-fatal: on error → warn + return vec![] (never block the turn).
    /// Content is formatted for human readability but OPAQUE to AgentLoop logic.
    async fn context_for_turn(&self, _owner: &str, _turn_msg: &str) -> Vec<ContextBlock> {
        let store = self.slice_store.read().await;
        if store.is_empty() {
            return vec![];
        }

        let mut blocks = Vec::new();
        for (from_owner, beliefs) in store.iter() {
            if beliefs.is_empty() {
                continue;
            }
            // Format as opaque text block — AgentLoop includes verbatim, never parses structure
            let lines: Vec<String> = beliefs.iter().map(|b| {
                let tag = b.persona_tag.as_deref().unwrap_or("general");
                format!("[{}:{}] {}", from_owner, tag, b.content)
            }).collect();
            let content = format!(
                "=== Shared context from {} ===\n{}\n===",
                from_owner,
                lines.join("\n")
            );

            blocks.push(ContextBlock {
                content,
                // CloudOk: filter_for_mesh guarantees only CloudOk beliefs in the slice
                max_tier: PrivacyTier::CloudOk,
            });
        }
        blocks
    }
}
```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| age::decrypt result → from_owner claim | Decrypted payload from_owner must match envelope claim; mismatch = spoofing attempt |
| MeshSliceStore → system prompt | Remote content enters system prompt via SEAM #2; if AgentLoop parsed it, prompt injection risk |
| P2PTransport::send → remote peer_url | POST to a URL from bastion.toml; malicious peer_url could be a SSRF vector |
| MeshSliceProvider context_for_turn → build_system_prompt | Slice content is opaque; must NOT be parsed by AgentLoop |
